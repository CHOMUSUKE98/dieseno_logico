int led1 = 11;
int led2 = 12;
int led3 = 13;

void setup() {
  pinMode(led1, OUTPUT);
  pinMode(led2, OUTPUT);
  pinMode(led3, OUTPUT);

  randomSeed(analogRead(A0));
}

void loop() {

  digitalWrite(led1, LOW);
  digitalWrite(led2, LOW);
  digitalWrite(led3, LOW);

  int numero = random(1, 4);

  if (numero == 1) {
    digitalWrite(led1, HIGH);
  }

  if (numero == 2) {
    digitalWrite(led2, HIGH);
  }

  if (numero == 3) {
    digitalWrite(led3, HIGH);
  }

  delay(500);

}
